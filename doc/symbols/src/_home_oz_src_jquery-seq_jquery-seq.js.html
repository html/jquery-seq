1 /** 
  2  * @function eachStep
  3  * @description Map array collection to deferred callback. Callback executes on every element after previous execution was finished.
  4  * @example eachStep(["Hello world", "Goodbye world"], function(hello){
  5  *    var def = $.Deferred();
  6  *
  7  *    setTimeout(function(){
  8  *      alert(hello);
  9  *      def.resolve();
 10  *    }, 1000);
 11  *
 12  *    return def;
 13  * }, function(){alert("Just said good bye");});
 14  *
 15  * @param {Array} collection A collection of items to map one after another
 16  * @param {Function} callback A callback that will be called on every item from collection. Callback should return deferred object.
 17  * @param {Function} [endcallback] A callback that will be called after every collection element is mapped.
 18  */
 19 
 20 function eachStep(collection, callback, endcallback){
 21   if(collection.length == 0){
 22     return endcallback && endcallback();
 23   }
 24 
 25   jQuery.when(callback(collection[0])).always(function(){
 26     eachStep(collection.slice(1), callback, endcallback);
 27   });
 28 }

 30 /**
 31  * @function sequentialExecute
 32  * @description Executes array of deferred callbacks one after another.
 33  * @example sequentialExecute([
 34  *   function(){ 
 35  *     var def = $.Deferred()
 36  *
 37  *     setTimeout(function(){
 38  *       alert("Hello world");
 39  *       def.resolve();
 40  *     }, 1000);
 41  *
 42  *     return def;
 43  *   },
 44  *   function(){ 
 45  *     var def = $.Deferred()
 46  *
 47  *     setTimeout(function(){
 48  *       alert("Goodbye world");
 49  *       def.resolve();
 50  *     }, 1000);
 51  *
 52  *     return def;
 53  *   }], function(){ alert('All executed'); });
 54  *
 55  * @param {Array} collection A collection of functions (callbacks) to execute one after another.
 56  * @param {Function} [endcallback] A callback that will be called after all callbacks from collection.
 57  */
 58 function sequentialExecute(collection, endcallback){
 59   if(collection.length == 0){
 60     return endcallback && endcallback();
 61   }
 62 
 63   jQuery.when(collection[0]()).always(function(){
 64     sequentialExecute(collection.slice(1), endcallback);
 65   });
 66 }

 68 function getGenerator(eachStepCallback){
 69   return function(collection, callback){
 70     return eachStep(collection, eachStepCallback, callback);
 71   }
 72 }

 74 var cache = [];
 75 
 76 function cachingGetGenerator(eachStepCallback){
 77   var temporary = function(collection, endcallback){
 78     if(collection.length == 0){
 79       return endcallback && endcallback();
 80     }
 81 
 82     var script = collection[0];
 83     var func = temporary;
 84     var callback = function(){
 85       func(collection.slice(1), endcallback);
 86     };
 87 
 88     if(cache.indexOf(script) == -1){
 89       eachStepCallback(script, function(){
 90         cache.push(script);
 91         callback();
 92       });
 93     }else{
 94       callback();
 95     }
 96   }
 97 
 98   return temporary;
 99 }

101 function withGetGenerator(allStuffCallback){
102   return function withScripts(){
103     var old$ = window.$;
104     window.$ = jQuery;
105     var scripts = Array.prototype.slice.call(arguments, 0, arguments.length - 1);
106     var endcallback = arguments[arguments.length - 1];
107 
108     allStuffCallback(scripts, function(){
109       endcallback();
110       window.$ = old$;
111     });
112   };
113 }

115 var loadImageCache = {}
116 function loadImage(imageSrc) {
117   var deferred = jQuery.Deferred();
118   if (typeof loadImageCache[imageSrc] === "undefined") {
119 
120     preloader         = new Image();
121     preloader.onload  = function() { 
122       //console && console.log("Loaded image " + this.src);
123       deferred.resolve(this.src) 
124     };
125     preloader.onerror = function() { 
126       console && console.log("Can not load an image " + this.src);
127       deferred.reject(this.src)  
128     };
129     preloader.src     = imageSrc;
130 
131     loadImageCache[imageSrc] = true;
132   }else{
133     console && console.log("Image cached " + imageSrc);
134     deferred.resolve(imageSrc);
135   }
136 
137   return deferred;
138 };

140 function appendStyleSheet(css_file){
141   var html_doc = document.getElementsByTagName('head').item(0);
142   var css = document.createElement('link');
143   css.setAttribute('rel', 'stylesheet');
144   css.setAttribute('type', 'text/css');
145   css.setAttribute('href', css_file);
146   html_doc.appendChild(css);
147   return false;
148 }
149 
150 function loadStyleSheet(source, callback){
151   return jQuery.get(source, function(){
152     appendStyleSheet(source);
153     callback && callback();
154   });
155 }
